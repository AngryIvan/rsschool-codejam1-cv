const solutionArray = [];
//retrieving a, b, c from equation string
const abc = (equation.split('* x'));

const buffer = abc[1].split('^2');
let b = (buffer[1]).split(' ');
b = parseInt(b.join('')); //deleting space between minus sign and a number
let c = (abc[2]).split(' ');
